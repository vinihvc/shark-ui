import { Preview } from "@/components/preview";
import { CodeBlock } from "@/components/code-block";
import { Button } from "@/components/ui/button";
import { Download, Heart, Settings, User } from "lucide-react";

## Installation

```bash
pnpm add @ark-ui/react
```

## Usage

```tsx
import { Button } from "@/components/ui/button";
```
