## Installation

<ComponentInstallation componentName="toggle-group" >
Install the following dependencies:
  
```bash
pnpm add @ark-ui/react tailwind-variants
```
</ComponentInstallation>

## Usage

```tsx lineNumbers
import { 
  ToggleGroup, 
  ToggleGroupItem 
} from "@/components/ui/toggle-group"; 
```

```tsx lineNumbers
<ToggleGroup>
  <ToggleGroupItem value="1">Item 1</ToggleGroupItem>
  <ToggleGroupItem value="2">Item 2</ToggleGroupItem>
</ToggleGroup>
```
