## Usage

```tsx lineNumbers
import {
  Resizable,
  ResizablePanel,
  ResizableResizeTrigger,
} from "@/components/ui/resizable";
```

```tsx lineNumbers
<Resizable>
  <ResizablePanel id="1">Panel 1</ResizablePanel>
  <ResizableResizeTrigger id="1:2" withHandle />
  <ResizablePanel id="2">Panel 2</ResizablePanel>
</Resizable>
```
