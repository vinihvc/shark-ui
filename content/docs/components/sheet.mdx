---

import { Preview } from "@/components/preview";
import { CodeBlock } from "@/components/code-block";
import { Sheet } from "@/components/ui/sheet";

import { X, Menu, Settings, User, LogOut } from "lucide-react";

## Installation

```bash
pnpm add @ark-ui/react
```

## Usage

```tsx
import { Sheet } from "@/components/ui/sheet";
```
