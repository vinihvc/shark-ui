import { Preview } from "@/components/preview";
import { CodeBlock } from "@/components/code-block";
import { Popover } from "@/components/ui/popover";
import { Button } from "@/components/ui/button";
import { ChevronRightIcon, XIcon } from "lucide-react";

## Installation

```bash
pnpm add @ark-ui/react
```

## Usage

```tsx
import { Popover } from "@/components/ui/popover";
```

## Examples

### Basic

<Preview>
  <Popover.Root>
    <Popover.Trigger asChild>
      <Button variant="outline">
        Open Popover

  </Popover.Root>
</Preview>

<CodeBlock>
```tsx

import { Button } from "@/components/ui/button";

export function PopoverDemo() {
  return (
    <Popover.Root>
      <Popover.Trigger asChild>
        <Button variant="outline">
          Open Popover
          <Popover.Indicator>
            <ChevronRightIcon className="h-4 w-4" />
          </Popover.Indicator>
        </Button>
      </Popover.Trigger>
      <Popover.Positioner>
        <Popover.Content>
          <Popover.Title>Popover Title</Popover.Title>
          <Popover.Description>
            This is a popover with a title and description.
          </Popover.Description>
          <Popover.CloseTrigger asChild>
            <Button variant="ghost" size="icon" className="absolute right-2 top-2 h-6 w-6">
              <XIcon className="h-4 w-4" />
            </Button>
          </Popover.CloseTrigger>
        </Popover.Content>
      </Popover.Positioner>
    </Popover.Root>
  );
}
```
</CodeBlock>

The title element of the popover.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Popover.Description

The description element of the popover.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Popover.CloseTrigger

The close trigger element of the popover.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Popover.Arrow

The arrow element of the popover.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |
