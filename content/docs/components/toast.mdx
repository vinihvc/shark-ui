## Usage

```tsx lineNumbers
import { Toaster } from "@/components/ui/toast";
```

```tsx lineNumbers
const toaster = useToast();

// add the toaster to the layout
<Toaster toaster={toaster} />

// show a toast
toaster.create({
  title: "Show Info Toast",
  description: "This is a info toast",
  type: "error",
});
```
