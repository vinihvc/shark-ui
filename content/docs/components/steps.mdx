## Usage

```tsx lineNumbers
import { 
  Steps, 
  StepsList, 
  StepsItem, 
  StepsTrigger, 
  StepsSeparator, 
  StepsIndicator, 
  StepsContent, 
  StepsCompletedContent, 
  StepsNext, 
  StepsPrevious 
} from "@/components/ui/steps";
```

```tsx lineNumbers
<Steps>
  <StepsList>
    <StepsItem index={0}>
      <StepsTrigger>Step 1</StepsTrigger>
    </StepsItem>
  </StepsList>

  <StepsContent index={0}>Content 1</StepsContent>

  <StepsCompletedContent>Completed Content</StepsCompletedContent>

  <StepsNext>Next</StepsNext>
  <StepsPrevious>Previous</StepsPrevious>
</Steps>
```
