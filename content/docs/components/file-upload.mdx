## Usage

```tsx lineNumbers
import { 
  FileUpload,
  FileUploadTrigger,
  FileUploadItemGroup,
  FileUploadContext,
  FileUploadItem,
  FileUploadItemPreview,
  FileUploadItemPreviewImage,
} from "@/components/ui/file-upload";
```

```tsx lineNumbers
<FileUpload maxFiles={4}>
  <FileUploadTrigger>Upload</FileUploadTrigger>
  <FileUploadItemGroup>
    <FileUploadContext>
      {({ acceptedFiles }) => (
        <FileUploadItem file={acceptedFiles[0]}>
          <FileUploadItemPreview>
            <FileUploadItemPreviewImage src={acceptedFiles[0].name} />
          </FileUploadItemPreview>
        </FileUploadItem>
      )}
    </FileUploadContext>
  </FileUploadItemGroup>
</FileUpload>
```
