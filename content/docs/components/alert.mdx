## Installation

<ComponentInstallation componentName="alert" >
Install the following dependencies:

```bash
pnpm add @ark-ui/react tailwind-variants
```
</ComponentInstallation>

## Basic usage

```tsx lineNumbers
import { 
  Alert, 
  AlertDescription, 
  AlertTitle 
} from "@/components/ui/alert";
```

```tsx lineNumbers
<Alert>
  <AlertTitle>Alert</AlertTitle>
  <AlertDescription>This is an alert</AlertDescription>
</Alert>
```

## Variants

You can change the variant of the alert using the `variant` prop. 

Available variants are: `solid`, `info`, `warning`, `success`, `destructive`.

<ComponentVariants variants={['solid', 'info', 'success', 'warning', 'destructive']}>
  <TabsContent value="solid">
    <Alert variant="solid">
      <Info />
      <AlertTitle>Deployment successful</AlertTitle>
      <AlertDescription>
        You can now start building your next great project.
      </AlertDescription>
    </Alert>
  </TabsContent>

  <TabsContent value="info">
    <Alert variant="info">
      <Info />
      <AlertTitle>Deployment information</AlertTitle>
      <AlertDescription>
        You can now start building your next great project.
      </AlertDescription>
    </Alert>
  </TabsContent>

  <TabsContent value="warning">
    <Alert variant="warning">
      <AlertTriangle />
      <AlertTitle>Deployment warning</AlertTitle>
      <AlertDescription>
        You can now start building your next great project.
      </AlertDescription>
    </Alert>
  </TabsContent>

  <TabsContent value="success">
    <Alert variant="success">

      </AlertDescription>

  <TabsContent value="destructive">

      </AlertDescription>
    </Alert>
  </TabsContent>
