```bash
pnpm add @ark-ui/react
```

## Usage

```tsx lineNumbers
import { 
  Select,
  SelectTrigger,
  SelectValue,
} from "@/registry/react/components/select";
```

```tsx lineNumbers
  <Select>
    <SelectTrigger>
      <SelectValueText placeholder="Select an option" />
    </SelectTrigger>
    <SelectContent>
      <SelectItem value="1">Option 1</SelectItem>
    </SelectContent>
  </Select>
```
