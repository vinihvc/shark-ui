## Installation

```bash
pnpm add @ark-ui/react
```

## Usage

```tsx
import { Select } from "@/components/ui/select";
```

## Examples

### Basic

<Preview>
  <Select.Root collection={Select.createListCollection({ items: ['React', 'Solid', 'Vue', 'Svelte'] })}>
    <Select.Label>Framework</Select.Label>
    <Select.Control>
      <Select.Trigger>
        <Select.ValueText placeholder="Select a Framework" />
        <Select.Indicator />
      </Select.Trigger>
      <Select.ClearTrigger>Clear</Select.ClearTrigger>
    </Select.Control>
    <Select.Content>
      <Select.ItemGroup>
        <Select.ItemGroupLabel>Frameworks</Select.ItemGroupLabel>
        {['React', 'Solid', 'Vue', 'Svelte'].map((framework) => (
          <Select.Item key={framework} item={framework}>
            <Select.ItemText>{framework}</Select.ItemText>
            <Select.ItemIndicator />
          </Select.Item>
        ))}
      </Select.ItemGroup>
    </Select.Content>
    <Select.HiddenSelect />
  </Select.Root>
</Preview>

<CodeBlock>
```tsx
import { Select } from "@/components/ui/select";

export function SelectDemo() {
  const collection = Select.createListCollection({ items: ['React', 'Solid', 'Vue', 'Svelte'] });

  return (
    <Select.Root collection={collection}>
      <Select.Label>Framework</Select.Label>
      <Select.Control>
        <Select.Trigger>
          <Select.ValueText placeholder="Select a Framework" />
          <Select.Indicator />
        </Select.Trigger>
        <Select.ClearTrigger>Clear</Select.ClearTrigger>
      </Select.Control>
      <Select.Content>
        <Select.ItemGroup>
          <Select.ItemGroupLabel>Frameworks</Select.ItemGroupLabel>
          {collection.items.map((item) => (
            <Select.Item key={item} item={item}>
              <Select.ItemText>{item}</Select.ItemText>
              <Select.ItemIndicator />
            </Select.Item>
          ))}
        </Select.ItemGroup>
      </Select.Content>
      <Select.HiddenSelect />
    </Select.Root>
  );
}
```
</CodeBlock>

The indicator element of the select.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Select.ClearTrigger

The clear trigger element of the select.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Select.Content

The content element of the select.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Select.ItemGroup

A group of select items.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Select.ItemGroupLabel

The label for a group of select items.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Select.Item

An individual select item.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `item` | `any` | - | The item data. |
| `disabled` | `boolean` | - | Whether the item is disabled. |

### Select.ItemText

The text element of the select item.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Select.ItemIndicator

The indicator element of the select item.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |

### Select.HiddenSelect

The hidden select element for form submission.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `asChild` | `boolean` | - | Use the provided child element as the default rendered element. |
