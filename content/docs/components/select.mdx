## Installation

<ComponentInstallation componentName="select" >
Install the following dependencies:
  
```bash
pnpm add @ark-ui/react lucide-react tailwind-variants
```
</ComponentInstallation>

## Usage

```tsx lineNumbers
import { 
  Select,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
```

```tsx lineNumbers
  <Select>
    <SelectTrigger>
      <SelectValueText placeholder="Select an option" />
    </SelectTrigger>
    <SelectContent>
      <SelectItem value="1">Option 1</SelectItem>
    </SelectContent>
  </Select>
```
