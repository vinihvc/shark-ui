## Usage

```tsx lineNumbers
import { 
  Card, 
  CardContent, 
  CardDescription, 
  CardFooter, 
  CardHeader, 
  CardTitle, 
  CardActions,
} from "@/components/ui/card";
```

```tsx lineNumbers
<Card>
  <CardHeader>
    <CardTitle />
    <CardDescription />
    <CardActions />
  </CardHeader>
  <CardContent />
  <CardFooter />
</Card>
```
