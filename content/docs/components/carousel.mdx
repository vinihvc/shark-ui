## Usage

```tsx lineNumbers
import { 
  Carousel,
  CarouselControl,
  CarouselItem,
  CarouselItemGroup,
  CarouselIndicator,
  CarouselIndicatorGroup,
  CarouselNext,
  CarouselPrevious,
} from "@/components/ui/carousel"; 
```

```tsx lineNumbers
<Carousel>
  <CarouselControl>
    <CarouselPrevious>Previous</CarouselPrevious>
    <CarouselNext>Next</CarouselNext>
  </CarouselControl>

  <CarouselItemGroup>
    <CarouselItem />
  </CarouselItemGroup>
</Carousel>
```
