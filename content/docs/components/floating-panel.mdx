## Installation

<ComponentInstallation componentName="floating-panel">
Install the following dependencies:
  
```bash
pnpm add @ark-ui/react lucide-react
```
</ComponentInstallation>

## Usage

```tsx lineNumbers
import { 
  FloatingPanel,
  FloatingPanelTrigger,
  FloatingPanelContent,
  FloatingPanelHeader,
  FloatingPanelTitle,
  FloatingPanelBody,
} from "@/components/ui/floating-panel";
```

```tsx lineNumbers
<FloatingPanel>
  <FloatingPanelTrigger />
  <FloatingPanelContent>
    <FloatingPanelHeader>
      <FloatingPanelTitle />
    </FloatingPanelHeader>
    <FloatingPanelBody />
  </FloatingPanelContent>
</FloatingPanel>
