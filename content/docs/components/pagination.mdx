## Installation

<ComponentInstallation componentName="pagination" >
Install the following dependencies:
  
```bash
pnpm add @ark-ui/react lucide-react
```
</ComponentInstallation>

## Usage

```tsx lineNumbers
import { 
  Pagination,
  PaginationContext,
  PaginationEllipsis,
  PaginationItem,
  PaginationNext,
  PaginationPrevious,
} from "@/components/ui/pagination";
```

```tsx lineNumbers
  <Pagination>
    <PaginationPrevious />
    <PaginationContext>
      <PaginationItem />
      <PaginationEllipsis />
    </PaginationContext>
    <PaginationNext />
  </Pagination>
```
