## Installation

```bash
npm install @ark-ui/react lucide-react
```

## Usage

```tsx
import { JsonTreeView } from "@/components/ui/json-tree-view";

const data = { name: "John", age: 30, address: { city: "NYC" } };

<JsonTreeView data={data} defaultExpandedDepth={1} />
```
